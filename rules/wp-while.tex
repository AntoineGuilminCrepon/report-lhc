\infer*[lab=wp-while, Right={$I \cap J = \emptyset$}]{
    \V
    P
    |- 
    \WPU {\m[i: g_i | i \in I]}
    {\WPE {\m[j: g_j | j \in J]}
    {\fun b. (b =_{I\cup J} 0 \land Q) \lor
    (b \ne_{I\cup J} 0 \land
    \WPU {\m[i: t_i | i \in I]}{\WPE {\m[j: t_j' | j \in J]} {P}})
    }
    }
}{
    \V
    {P}
    |- 
    \WPU {\m[i: \code{while}\ g_i\ \code{do}\ t_i | i \in I]}
    {\WPE {\m[j: \code{while}\ g_j\ \code{do}\ t_j' | j \in J]} {Q}}
}
