\infer*[lab=wp-while]{
    \V
    P
    |- 
    \WPU {\m[i: g_i | i \in I]}
    {\fun b. (b =_{I} 0 \land Q) \lor
    (b \ne_{I} 0 \land
    \WPU {\m[i: t_i | i \in I]}{\WPE {\m[i: t_i' | i \in I]} {P}})
    }
}{
    \V
    {P}
    |- 
    \WPU {\m[i: \code{while}\ g_i\ \code{do}\ t_i | i \in I]}
    {\WPE {\m[i: \code{while}\ g_i\ \code{do}\ t_i' | i \in I]} {Q}}
}
